1️⃣ Create Backend API: Analyze All Vulnerabilities

Add this route in your routes.ts.

app.post("/api/analyzeall", async (req, res) => {
  try {
    const vulnerabilities = await storage.getVulnerabilities();

    if (vulnerabilities.length === 0) {
      return res.status(400).json({
        error: "No vulnerabilities available. Collect data first."
      });
    }

    let analyzedCount = 0;
    let alertsCreated = 0;

    for (const vuln of vulnerabilities) {
      // Predict risk
      const prediction = predictRisk(vuln);
      await storage.addPrediction(prediction);

      // Generate mitigation
      const mitigationResult = await generateMitigationPlan(
        vuln.cveId,
        vuln.vulnerabilityName,
        vuln.shortDescription,
        prediction.riskLevel,
        prediction.probability
      );

      const mitigation = {
        cveId: vuln.cveId,
        riskLevel: prediction.riskLevel,
        urgency: mitigationResult.urgency,
        mitigation: mitigationResult.mitigation,
        steps: mitigationResult.steps,
        generatedAt: new Date().toISOString()
      };

      await storage.addMitigation(mitigation);

      // Create alert if high risk
      if (prediction.riskLevel === "High" && prediction.probability >= 0.8) {
        await storage.addAlert({
          cveId: vuln.cveId,
          riskLevel: prediction.riskLevel,
          probability: prediction.probability,
          mitigation: mitigation.mitigation,
          status: "pending",
          triggeredBy: "bulk_analysis"
        });

        alertsCreated++;
      }

      analyzedCount++;
    }

    await storage.addAuditLog({
      action: "Bulk Vulnerability Analysis",
      user: "system",
      details: `Analyzed ${analyzedCount} vulnerabilities. Alerts created: ${alertsCreated}`,
      category: "analysis"
    });

    res.json({
      success: true,
      analyzed: analyzedCount,
      alertsCreated
    });

  } catch (error) {
    console.error("Bulk analysis error:", error);
    res.status(500).json({ error: "Failed to analyze vulnerabilities" });
  }
});

Yes 👍 — after alerts are generated you can automatically send them to a user via email. This is very common in vulnerability management systems.

Typical flow:

Vulnerability analyzed
        ↓
High-risk alert created
        ↓
Store alert in database
        ↓
Send email notification

You already create alerts here in your backend:

Right after this, you can send an email.

1️⃣ Install Email Library

The easiest Node.js email library is:

Nodemailer

Install:

npm install nodemailer
2️⃣ Create Email Utility

Create a file:

server/lib/email.ts
import nodemailer from "nodemailer";

const transporter = nodemailer.createTransport({
  service: "gmail",
  auth: {
    user: process.env.ALERT_EMAIL,
    pass: process.env.ALERT_EMAIL_PASSWORD
  }
});

export async function sendAlertEmail(
  email: string,
  cveId: string,
  riskLevel: string,
  mitigation: string
) {
  const mailOptions = {
    from: process.env.ALERT_EMAIL,
    to: email,
    subject: `🚨 Security Alert: ${cveId}`,
    html: `
      <h2>High Risk Vulnerability Detected</h2>
      <p><strong>CVE:</strong> ${cveId}</p>
      <p><strong>Risk Level:</strong> ${riskLevel}</p>

      <h3>Recommended Mitigation</h3>
      <p>${mitigation}</p>

      <p>Please review immediately.</p>
    `
  };

  await transporter.sendMail(mailOptions);
}
